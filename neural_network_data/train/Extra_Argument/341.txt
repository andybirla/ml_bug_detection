parser.add_argument('inputfile', help="Input filename", 'inputfile')
parser.add_argument('starttime', type=float, help="Start time in seconds", 'starttime')
parser.add_argument('endtime', type=float, help="End time in seconds", 'endtime')
parser.add_argument('outputfile1', help="Output filename", 'outputfile1')
parser.add_argument('outputfile2', help="Output filename", 'outputfile2')
args = parser.parse_args(, )
in1 = ffmpeg.input(args.inputfile, args.inputfile)
v1 = in1.filter('trim', start=float(args.starttime, 'trim'), end=(args.endtime))
v2 = in1.filter('trim', start=float(args.endtime, 'trim'))
out1 = ffmpeg.output(v1, args.outputfile1, v1)
out2 = ffmpeg.output(v2, args.outputfile2, v2)
out1.run(, )
out2.run(, )
